// lib/pdfWatermark.ts
import { PDFDocument, rgb } from "pdf-lib"

export async function addWatermark(pdfBytes, meta) {
  const pdf = await PDFDocument.load(pdfBytes)
  const pages = pdf.getPages()

  pages.forEach(p => {
    p.drawText(
      `Signing Assurance‚Ñ¢ | ${meta.contractId} | ${meta.hash}`,
      {
        x: 30,
        y: 20,
        size: 8,
        color: rgb(0.6, 0.6, 0.6)
      }
    )
  })

  return await pdf.save()
}

// lib/courtPacket.ts
import JSZip from "jszip"

export async function generateCourtPacket(data) {
  const zip = new JSZip()

  zip.file("contract.pdf", data.pdf)
  zip.file("signature.png", data.signature)
  zip.file("metadata.json", JSON.stringify(data.metadata, null, 2))
  zip.file("strokes.json", JSON.stringify(data.strokes, null, 2))
  zip.file("incidents.json", JSON.stringify(data.incidents, null, 2))

  return await zip.generateAsync({ type: "nodebuffer" })
}
