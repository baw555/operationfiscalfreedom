type DegradedFeatureNoticeProps = {
  featureName: string;
  description: string;
  onProceed: () => void;
  onReport: () => void;
};

export function DegradedFeatureNotice({
  featureName,
  description,
  onProceed,
  onReport,
}: DegradedFeatureNoticeProps) {
  return (
    <div className="rounded border border-yellow-400 bg-yellow-50 p-4">
      <h3 className="font-semibold text-yellow-800">
        {featureName} Unavailable
      </h3>

      <p className="mt-1 text-sm text-yellow-700">
        {description}
      </p>

      <div className="mt-4 flex gap-3">
        <button
          onClick={onProceed}
          className="rounded bg-yellow-600 px-3 py-1.5 text-white hover:bg-yellow-700"
        >
          Proceed Anyway
        </button>

        <button
          onClick={onReport}
          className="rounded border border-yellow-600 px-3 py-1.5 text-yellow-700 hover:bg-yellow-100"
        >
          Proceed & File Report
        </button>
      </div>
    </div>
  );
}

2Ô∏è‚É£ HOW TO USE IT (example: SignaturePanel)
function SignaturePanel() {
  const { setCapabilities, setDegradedReports } = useNdaContext();

  const cameraUnavailable = /* detected failure */;

  if (cameraUnavailable) {
    return (
      <DegradedFeatureNotice
        featureName="Camera Signature"
        description="Camera access failed or was denied. You may continue using a typed signature."
        onProceed={() => {
          setCapabilities((c) => ({
            ...c,
            camera: "unavailable",
          }));
        }}
        onReport={() => {
          setCapabilities((c) => ({
            ...c,
            camera: "unavailable",
          }));

          setDegradedReports((r) => [
            ...r,
            {
              feature: "camera",
              reason: "permission_denied",
              timestamp: Date.now(),
            },
          ]);
        }}
      />
    );
  }

  return <ActualSignatureCanvas />;
}
