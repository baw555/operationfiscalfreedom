// components/SignaturePad.tsx (extended)
"use client"
import { useRef } from "react"

export function SignaturePad({ onSave }) {
  const canvasRef = useRef<HTMLCanvasElement>(null)
  const strokes: any[] = []
  let drawing = false
  let currentStroke: any = null

  function start(e) {
    drawing = true
    currentStroke = {
      points: [],
      startTime: Date.now()
    }
  }

  function draw(e) {
    if (!drawing) return
    const ctx = canvasRef.current!.getContext("2d")!
    ctx.lineWidth = 2
    ctx.lineCap = "round"

    const point = {
      x: e.nativeEvent.offsetX,
      y: e.nativeEvent.offsetY,
      t: Date.now()
    }

    currentStroke.points.push(point)
    ctx.lineTo(point.x, point.y)
    ctx.stroke()
  }

  function stop() {
    if (!drawing) return
    drawing = false
    currentStroke.endTime = Date.now()
    strokes.push(currentStroke)
  }

  function save() {
    const dataUrl = canvasRef.current!.toDataURL("image/png")
    onSave({
      image: dataUrl,
      strokeMetadata: strokes,
      canvas: {
        width: 500,
        height: 200
      }
    })
  }

  return (
    <div>
      <canvas
        ref={canvasRef}
        width={500}
        height={200}
        className="border"
        onMouseDown={start}
        onMouseMove={draw}
        onMouseUp={stop}
      />
      <button onClick={save} className="mt-2">
        Save Signature
      </button>
    </div>
  )
}
