7.1 Rule-based suggestions (explainable)

src/lib/strengthSuggestions.ts

export function suggestImprovements(heatmap: any[]) {
  const suggestions: string[] = [];

  for (const h of heatmap) {
    if (h.avgStrength < 3 && h.type === "medical") {
      suggestions.push(
        `Consider obtaining updated medical records or treatment notes for ${h.condition}.`
      );
    }

    if (h.avgStrength < 3 && h.type === "nexus") {
      suggestions.push(
        `A medical opinion connecting ${h.condition} to service could materially strengthen this claim.`
      );
    }

    if (h.type === "lay" && h.avgStrength < 3) {
      suggestions.push(
        `Additional personal or buddy statements describing functional impact of ${h.condition} may help.`
      );
    }
  }

  return [...new Set(suggestions)];
}

export function buildVendorScorecard(actions: any[]) {
  const uploads = actions.filter(a => a.type === "upload").length;
  const notes = actions.filter(a => a.type === "note").length;
  const avgResponse =
    actions.length
      ? Math.round(
          actions.reduce((a,b)=>a+b.responseTimeHours,0)/actions.length
        )
      : null;

  return {
    uploads,
    notes,
    avgResponseHours: avgResponse,
    score:
      uploads * 2 +
      notes +
      (avgResponse !== null ? Math.max(10 - avgResponse, 0) : 0)
  };
}

STAGE 9 — VA LANE CONFIDENCE PERCENTAGE

(Decision support, not decision making)

Purpose

Veterans want to know:

“How confident is this recommendation?”

9.1 Confidence model (transparent)

src/lib/laneConfidence.ts

export function laneConfidence({
  strengthAvg,
  completenessPct,
  hasNewEvidence
}: {
  strengthAvg: number;
  completenessPct: number;
  hasNewEvidence: boolean;
}) {
  let confidence = 50;

  confidence += (strengthAvg - 3) * 15;
  confidence += (completenessPct - 70) * 0.5;
  if (hasNewEvidence) confidence += 10;

  return Math.max(10, Math.min(95, Math.round(confidence)));
}
